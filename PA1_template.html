Imputing missing values

total_missing <- sum(is.na(activity$steps))
total_missing
## [1] 2304
activity_filled <- activity

for(i in 1:nrow(activity_filled)) {
  if(is.na(activity_filled$steps[i])) {
    this_interval <- activity_filled$interval[i]
    interval_mean <- interval_avg$steps[interval_avg$interval == this_interval]
    activity_filled$steps[i] <- interval_mean
  }
}

Are there differences in activity patterns between weekdays and weekends?

activity_filled$day_type <- ifelse(weekdays(activity_filled$date) %in% c("Saturday", "Sunday"),
                                   "weekend",
                                   "weekday")

activity_filled$day_type <- as.factor(activity_filled$day_type)
